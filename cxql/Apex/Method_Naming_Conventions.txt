// This query reports any method declaration from .cls files that is not camel case
CxList methodDecls = Find_MethodDecls().FindByFileName("*.cls");

string regex = @"\s*[a-z][a-zA-Z0-9]*\s*[(]";
CxList nameFollowingConventions = All.FindByRegexExt(
	regex, 
	"*.cls", 
	false, 
	System.Text.RegularExpressions.RegexOptions.Singleline
);

List<string> validNames = nameFollowingConventions.CxSelectElementValues<CSharpGraph, string>(
	x => x.ShortName.Replace("(", "").Trim()
);

CxList methodDeclsOfCamelCase = methodDecls.FindByShortNames(validNames, false);
result = methodDecls - methodDeclsOfCamelCase;
